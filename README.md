## Multi-Threading

```python

def av_arduino(coms):
    av_coms = []
    while True:
        for com in coms:
            ser.port = com
            if ser.is_open == True:
                continue
            else:
                av_coms.append(com)
        if av_coms:
            break
        else:
            time.sleep(5)
            continue
    av_com = random.choice(av_coms)
    return av_com
    
ssids = ["MOYOXXL 2.0", "_ONOWiFi", "_ONOWiFiXXX"]
for ssid in ssids:
    av_com = av_arduino(coms)
    print("Random:", av_com)
    if av_com:
        thread_with_args = Thread(target=use_arduino, args=(av_com, ssid))
        thread_with_args.start()
        thread_with_args.join()
```
